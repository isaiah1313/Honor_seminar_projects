\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{特殊相対性理論の出発点}
        \begin{itemize}
            \item 全ての慣性系\footnote{力が加わらない物体が静止あるいは等速直線運動をするように見える系のこと}で物理法則は同じ形をしている
            \item 相互作用の伝達速度には上限が存在する
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{ミンコフスキー空間 $\bm{M}$}
        時間1次元と空間3次元を統合した4次元実ベクトル空間 $\bm{M}$ を考える。
        \begin{itemize}
            \item 位置と時間をまとめて \textbf{4元位置ベクトル} $x \in \bm{M}$ で表す：
            \begin{equation*}
                x = [x^\mu] = [ct, x, y, z]^T=[ct,\bm{r}] \quad (\mu = 0, 1, 2, 3)
            \end{equation*}
            \item 質点の運動はミンコフスキー空間上では曲線として表される。
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{特殊相対性理論の舞台}
    特殊相対性理論で考える座標変換は
    \begin{itemize}
        \item 同じ慣性系での座標軸変換
        \item 異なる速度で動く慣性系との座標変換
    \end{itemize}
    の2種類である。これらの座標変換をまとめてローレンツ変換と呼ぶ。\\
    この座標変換は線形であり，2階テンソルを用いて表現することができる。
\end{frame}

\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{ローレンツ変換}
        \begin{itemize}
            \item 元位置ベクトル$x^\mu$を異なる慣性系から見た時の4元位置ベクトル$x'^\mu$はローレンツ変換テンソル$\Lambda=\tensor{\Lambda}{_\mu_\nu}$を用いて
            \begin{align*}
                x'^\mu = x^\nu \tensor{\Lambda}{^\mu_\nu}
            \end{align*}
        と求められる。\\
        \item ローレンツ変換の前後で$g=\mathrm{diag}(+1,-1,-1,-1)$が変化しない:
            \begin{align*}
                \tensor{g}{_\mu_\nu}&=\tensor{g}{_\rho_\sigma}\tensor{\Lambda}{^\rho_\mu}\tensor{\Lambda}{^\sigma_\nu}\\
            \end{align*}
        \end{itemize}
    \end{block}    
\end{frame}

\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{計量$g$}
        \begin{itemize}
            \item 4元ベクトル $a \in \bm{M}$ の"大きさ"を計量 $g = \mathrm{diag}(1, -1, -1, -1)$ を用いて定義する：
            \begin{equation*}
                \norm{a}^2 =  \tensor{g}{_\mu_\nu} a^\mu a^\nu  = (a_0)^2-(a_1)^2-(a_2)^2-(a_3)^2
            \end{equation*}
            \item ローレンツ変換の前後で4元ベクトルの"大きさ"は変化しない:
            \begin{align*}
                \norm{a}^2=\tensor{g}{_\mu_\nu} a^\mu a^\nu&=\tensor{g}{_\mu_\nu} a'^\mu a'^\nu=\norm{a'}^2\\
                \iff
                \tensor{g}{_\mu_\nu}&=\tensor{g}{_\rho_\sigma} \tensor{\Lambda}{^\rho_\mu}\tensor{\Lambda}{^\sigma_\nu}
            \end{align*}
            
        \end{itemize}        
    \end{block}
\end{frame}

\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{双対空間$\bm{M}^*$}
        \begin{itemize}
        \item $a^\mu\in\bm{M}$を計量$g$を用いて添字を下げる:
            \begin{align*}
                a_\nu=\tensor{g}{_\mu_\nu} a^\mu
            \end{align*}
            下付き添字のベクトルの集合を$M$の双対空間と呼び$\bm{M}^*$と書く。
        \item $\tensor{g}{_\mu_\nu}$の逆写像として上付きの計量$\tensor{g}{^\mu^\nu}$を定義する。:
        \begin{align*}
            a^\mu=\tensor{g}{^\mu^\nu}a_\nu
        \end{align*}
        \end{itemize}
    \end{block}
これによって$\bm{M}$と$\bm{M}^*$に1対1の対応関係を作ることができた。\\
$a^\mu$を反変ベクトル，$a_\mu$を共変ベクトルと呼ぶ
\end{frame}


\begin{frame}{特殊相対性理論の舞台}
    \begin{block}{相対論的な物理の記述}
        ミンコフスキー空間$M$とその双対空間$M^*$を用いて物理を記述していく\\
        物理では基底を明記せずに係数だけを書くことが多い
        \begin{itemize}
            \item スカラー $s$
            \item 反変ベクトル$a^\mu\in\bm{M}$と共変ベクトル$a_\mu\in\bm{M}$
            \item 2階テンソル$\tensor{T}{^\mu^\nu}\in\bm{M}\otimes\bm{M}$,\quad$\tensor{T}{_\mu_\nu}\in\bm{M}^*\otimes\bm{M}^*$,\quad\dots
            \item n階テンソル$\tensor{T}{^{\mu_1}^{\mu_2}^{ \dots}^{\mu_n}}\in \bm{M}\otimes\bm{M}\dots \otimes\bm{M}$,\quad\dots
        \end{itemize}
    に物理量を対応させる
    \end{block}
\end{frame}

\section{最小作用の原理とは}
\begin{frame}{最小作用の原理とは}
    \begin{block}{ラグラジアン$L$}
        系の状態によって定まるエネルギーの次元を持つ量である\\
    \end{block}
ニュートン力学では$L(x,v)=\frac{1}{2}m v^2 -U(x)$と書けることが多い。
    \begin{block}{作用$S$}
        系の各瞬間のラグラジアンの値を時間積分した値：
        \begin{align*}
            S = \int_{t_0}^{t_1} L \>\mathrm{d}t
        \end{align*}
        系の時間発展が定まると作用$S$の値も一意に定まる
    \end{block}
\end{frame}


\begin{frame}{最小作用の原理とは}
    \begin{block}{最小作用の原理}
        系が時間発展するとき，実現する時間発展の作用の変分
        \footnote{作用の変分:系のある時間発展と，少しだけ変化させた系の時間発展での作用の差のこと}
        は$\delta S=0$となる。
    \end{block}
ニュートン力学の場合は最小作用の原理から運動方程式(オイラーラグランジュ方程式)を導出できる。
\begin{align*}
    0 = \delta S
    &= \int_{t_0}^{t_1}L(x+\delta x, v + \delta v)\>\mathrm{d}t - \int_{t_0}^{t_1} L(x,v) \>\mathrm{d}t \\
    &= \int_{t_0}^{t_1} \delta x \left[\pdiff{L}{x}-\diff{}{t}\pdiff{L}{v}\right]\>\mathrm{d}t
\end{align*}
\end{frame}

\begin{frame}{最小作用の原理とは}
    \begin{block}{作用，ラグラジアンの選び方}
        最小作用の原理の結果，現実世界を再現できるようなものであるなら作用はどのようなものを選んでも良い。\\
        一般な系決める方法は存在しないが，
        \begin{itemize}
            \item 系の対称性
            \item 作用の次元が$\mathrm{ML^2T^{-1}}$になる
            \item 作用がローレンツスカラーである\footnote{相対性原理から作用がローレンツスカラーであることが導ける}
        \end{itemize}
        などの要求からある程度の任意性を除いてラグラジアンや作用を定めることができる。 %ここは怪しい
    \end{block}
\end{frame}

\begin{frame}{質点の時間発展}
    \begin{block}{質点の運動の表現方法}
        質点の時間発展は$x^\mu(s)$と表すことができた。\\
        この時，$u^\mu(s)=\diff{x^\mu}{s}$はミンコフスキー空間での"速度"を表す曲線の接ベクトルになっている。$u^\mu(s)$のノルムは
        \begin{align*}
            \norm{u}^2=\tensor{g}{_\mu_\nu}u^\mu u^\nu=\tensor{g}{_\mu_\nu}\diff{x^\mu}{s}\diff{x^\nu}{s}=\frac{\tensor{g}{_\mu_\nu}\mathrm{d}x^\mu \mathrm{d}x^\nu}{\mathrm{d}s^2}=\frac{\mathrm{d}s^2}{\mathrm{d}s^2}=1
        \end{align*}
    \end{block}
\end{frame}


\begin{frame}{質点の時間発展}

        \item 等速直線運動が再現される
        \item ローレンツスカラーである
    \end{itemize}
    といった条件が科される。
\end{frame}

\begin{frame}{質点の時間発展}
    \begin{block}{時空間の回転対称性}
        時空間の回転対称の表現方法を考える。\\
        世界線$x^\mu(s)$をローレンツ変換テンソル$\tensor{\Lambda}{^\mu_\nu}$で変換させる：
        \begin{align*}
            x'^\mu=\tensor{\Lambda}{^\mu_\nu}x^\nu
        \end{align*}
        この変換によって全ての4元ベクトルは変換されるがノルムは変化しない：
        \begin{align*}
            \norm{x}=\norm{x'}
        \end{align*}
        よって時空間の回転対称性から\\
        作用は4元ベクトル自体は現れず，そのノルムのみに依存するとわかる。
    \end{block}
\end{frame}

\begin{frame}{質点の時間発展}
    作用として経路長$\int \mathrm{d}s$に比例する量を選べば，作用の停留条件は等速直線運動になると考えられる。\\
    次元調整と後のニュートン力学との整合性のために$-mc$をかけて，
\begin{columns}[T]
    \column{0.5\textwidth}

        S
        &=-mc\int_{A}^{B}\>\mathrm{d}s\\
        &=-mc\int_{A}^{B} \sqrt{c^2\mathrm{d}t^2-\mathrm{d}\bm{r}^2}\\
        &=-mc^2\int_{t_A}^{t_B}\sqrt{1-\left(\frac{\bm{v}}{c}\right)^2}\>\mathrm{d}t
    \end{align*}
    \column{0.5\textwidth}

\end{frame}

\begin{frame}{質点の時間発展}
    よって，相対論的な自由粒子のラグラジアンは
    \begin{align*}
        L(\bm{r},\bm{v})
        =-mc^2\sqrt{1-\left(\frac{\bm{v}}{c}\right)^2}
    \end{align*}
    であることがわかった。ここから最小作用の原理から導出されるオイラーラグランジュ方程式に代入すると運動方程式：
    \begin{align*}
        \diff{}{t}\left(\frac{\bm{v}}{\sqrt{1-\left(\frac{\bm{v}}{c}\right)^2}}\right)=0
    \end{align*}
    を得ることができる。
\end{frame}
